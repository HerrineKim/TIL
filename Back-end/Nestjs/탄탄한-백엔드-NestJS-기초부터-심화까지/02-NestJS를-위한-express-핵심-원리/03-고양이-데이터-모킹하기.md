# 고양이 데이터 모킹하기

# data 보내기 연습

**app.ts**

```typescript
// 고양이 데이터 모킹하기
const data = [1, 2, 3, 4]

app.get('/', (req: express.Request, res: express.Response) => {
  console.log(req)
  // key와 value가 같으면 하나만 써도 됨
  res.send({data})
} )

// listen(): 서버를 실행하는 메서드
// app.listen(port, () => console.log(`Example app listening on port http://localhost:${port}`))
app.listen(8000, () => {
  console.log('server is on...')
})
```
